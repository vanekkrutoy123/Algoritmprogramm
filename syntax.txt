  age: 30        // под ключом "age" хранится значение 30
};

// получаем свойства объекта:
alert( user.name ); // John
alert( user.age ); // 30

delete user.age;

      < - условие верно если первый оперант меньше второго
      > - условие верно если первый оперант больше второго
      <= - условие верно если первый оперант меньше или равен второму
      >= - условие верно если первый оперант больше или равен второму

      != - не равно

Оператор BREAK - полностью прекращает выполнение цикла и передаёт управление на блок кода за его телом 
Оператор CONTINUE (продолжение) - При выполнении оператора, цикл не прерывается, а переходит к следующей итерации( если условие всё еще равно true. ), этот цикл используют если понятно что на повторе текущего цикла делать уже нечего.
Пример: напишите программу с break и continue чтобы вывести только нечётные значения.
for (let i = 0; i < 10; i++){
        if (i % 2 !== 0) continue;
        alert (i)
}

for (let i = 0; i < 10; i++){
        if (i % 2 === 0) continue;
        alert (i)
}

Конструкция SWITCH, CASE И DEFAULT - используется для выполнения различных действий в зависимости от значения выражения она представляет собой альтернативу цепочки if, else. И обычно используется когда нужно сравнивать одно значение с множеством вариантов.
if, else проверяет многие факторы, а switch и case рассматривает основное

Если число равно 3, то выводим результат "МАЛО"
Если число ранво 4, то выводим результат "В ТОЧКУ"
Если число равно 5, то выводим результат "МНОГО"
Если число РАВНО != 3,4,5 то выводим результат "МИМО"

let a = +prompt("Введите число")

switch (a) {
 case 3:
                alert("Мало")
                break;
 case 4: 
                alert ("В точку")
                break;
 case 5:
                alert ("Много")
                break;
                default:
                        alert("Мимо")
                break;
}

function switchcase(num1) { 
     num1 = +prompt("введите число")
     switch(num1) { 
    case 0:
         alert('0')
         break
         case 1:
            alert('1')
             break
             case 2:
          case 3:
     alert('2 3') 
     break
     default:
     alert("не в диапозоне")
}
}
